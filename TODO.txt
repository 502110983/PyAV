- getting autogenerated Cython headers:

	pip install cython-codegen ctypeslib
	git clone git://github.com/gccxml/gccxml.git
	cd gccxml
	cmake .
	make
	cd ..
	ffmpeg=/usr/local/Cellar/ffmpeg/1.0/include
	gccxml/bin/gccxml $(pkg-config --cflags libavcodec) $ffmpeg/libavcodec/avcodec.h -o avcodec.xml
	xml2cython.py -l 'avcodec' $ffmpeg/libavcodec/avcodec.h avcodec.xml
